00105 /*
00106  * alt_irq_disable_all() 
00107  *
00108  * This routine inhibits all interrupts by negating the status register PIE 
00109  * bit. It returns the previous contents of the CPU status register (IRQ 
00110  * context) which can be used to restore the status register PIE bit to its 
00111  * state before this routine was called.
00112  */
00113 static ALT_INLINE alt_irq_context ALT_ALWAYS_INLINE 
00114        alt_irq_disable_all (void)
00115 {
00116   alt_irq_context context;
00117 
00118   NIOS2_READ_STATUS (context);
00119 
00120   NIOS2_WRITE_STATUS (context & ~NIOS2_STATUS_PIE_MSK);
00121   
00122   return context;
00123 }

00125 /*
00126  * alt_irq_enable_all() 
00127  *
00128  * Enable all interrupts that were previously disabled by alt_irq_disable_all()
00129  *
00130  * This routine accepts a context to restore the CPU status register PIE bit
00131  * to the state prior to a call to alt_irq_disable_all().
00132  
00133  * In the case of nested calls to alt_irq_disable_all()/alt_irq_enable_all(), 
00134  * this means that alt_irq_enable_all() does not necessarily re-enable
00135  * interrupts.
00136  *
00137  * This routine will perform a read-modify-write sequence to restore only
00138  * status.PIE if the processor is configured with options that add additional 
00139  * writeable status register bits. These include the MMU, MPU, the enhanced 
00140  * interrupt controller port, and shadow registers. Otherwise, as a performance
00141  * enhancement, status is overwritten with the prior context. 
00142  */
00143 static ALT_INLINE void ALT_ALWAYS_INLINE 
00144        alt_irq_enable_all (alt_irq_context context)
00145 {
00146 #if (NIOS2_NUM_OF_SHADOW_REG_SETS > 0) || (defined NIOS2_EIC_PRESENT) || \
00147     (defined NIOS2_MMU_PRESENT) || (defined NIOS2_MPU_PRESENT)
00148   alt_irq_context status;
00149   
00150   NIOS2_READ_STATUS (status);
00151   
00152   status &= ~NIOS2_STATUS_PIE_MSK;
00153   status |= (context & NIOS2_STATUS_PIE_MSK);
00154   
00155   NIOS2_WRITE_STATUS (status);
00156 #else
00157   NIOS2_WRITE_STATUS (context);
00158 #endif
00159 }
