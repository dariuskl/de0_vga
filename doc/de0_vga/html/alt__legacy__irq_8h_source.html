00102 /*
00103  * alt_irq_initerruptable() should only be called from within an ISR. It is used
00104  * to allow higer priority interrupts to interrupt the current ISR. The input
00105  * argument, "priority", is the priority, i.e. interrupt number of the current
00106  * interrupt.
00107  *
00108  * If this function is called, then the ISR is required to make a call to
00109  * alt_irq_non_interruptible() before returning. The input argument to
00110  * alt_irq_non_interruptible() is the return value from alt_irq_interruptible().
00111  *
00112  * Care should be taken when using this pair of functions, since they increasing
00113  * the system overhead associated with interrupt handling.
00114  *
00115  * If you are using an exception stack then nested interrupts won't work, so
00116  * these functions are not available in that case.
00117  */
00118 static ALT_INLINE alt_u32 ALT_ALWAYS_INLINE alt_irq_interruptible (alt_u32 priority)
00119 {
00120   extern volatile alt_u32 alt_priority_mask;
00121   extern volatile alt_u32 alt_irq_active;
00122 
00123   alt_u32 old_priority;
00124 
00125   old_priority      = alt_priority_mask;
00126   alt_priority_mask = (1 << priority) - 1;
00127 
00128   NIOS2_WRITE_IENABLE (alt_irq_active & alt_priority_mask);
00129 
00130   NIOS2_WRITE_STATUS (1);
00131 
00132   return old_priority; 
00133 }
