00040 /*
00041  * ALT_MODULE_CLASS
00042  *
00043  * This macro returns the module class name for the specified module instance.
00044  * It uses information in the system.h file.
00045  * Neither the instance name or class name are quoted (so that they can
00046  * be used with other pre-processor macros).
00047  *
00048  * Example:
00049  *   Assume the design has an instance of an altera_avalon_uart called uart1.
00050  *   Calling ALT_MODULE_CLASS(uart1) returns altera_avalon_uart.
00051  */
00052 
00053 #define ALT_MODULE_CLASS(instance) ALT_MODULE_CLASS_ ## instance
00054

00056 /*
00057  * ALT_DRIVER_FUNC_NAME
00058  *
00059  *   --> instance               Instance name.
00060  *   --> func                   Function name.
00061  *
00062  * This macro returns the device driver function name of the specified
00063  * module instance for the specified function name.
00064  *
00065  * Example:
00066  *   Assume the design has an instance of an altera_avalon_uart called uart1.
00067  *   Calling ALT_DRIVER_FUNC_NAME(uart1, write) returns 
00068  *   altera_avalon_uart_write.
00069  */
00070 
00071 #define ALT_DRIVER_FUNC_NAME(instance, func) \
00072   ALT_DRIVER_FUNC_NAME1(ALT_MODULE_CLASS(instance), func)
00073 #define ALT_DRIVER_FUNC_NAME1(module_class, func) \
00074   ALT_DRIVER_FUNC_NAME2(module_class, func)
00075 #define ALT_DRIVER_FUNC_NAME2(module_class, func) \
00076   module_class ## _ ## func

00079  * ALT_DRIVER_STATE_STRUCT
00080  *
00081  *   --> instance               Instance name.
00082  *
00083  * This macro returns the device driver state type name of the specified
00084  * module instance.
00085  *
00086  * Example:
